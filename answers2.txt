25. Menu script:
(#!/bin/bash)
menu_option_one() {
echo "What is the capital of Norway?"
echo "Oslo"
}

menu_option_two() {
echo "What is VIPS?"
echo "A tool for planning and documenting nursery"
}

menu_option_three() {
echo "What faculty is BWU in?"
echo "Faculty of architecture and design"
}

press_enter() {
echo ""
echo -n "	Press Enter to continue "
read
clear
}

incorrect_selection() {
echo "Wrong selection! Try again."
}

until ["selection" = "0"]; do
clear
echo ""
echo "	1 - What is the capital of Norway?"
echo "	2 - What is VIPS?"
echo "	3 - What faculty is BWU in?"
echo "	q - Exit"
echo -n "Enter selection: "
read selection
echo ""
case $selection in
1) clear ; menu_option_one ; press_enter ;;
2) clear ; menu_option_two ; press_enter ;;
3) clear ; menu_option_three ; press_enter ;;
q) clear ; exit ;;
*) clear ; incorrect_selection ; press_enter ;;
esac 
done
